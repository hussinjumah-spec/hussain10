/*
  ========================================================================
  FORMFLOW ULTIMATE SYNC & RESPONSE SCRIPT (V2 - CHUNKED)
  ========================================================================
  هذا الكود يدعم المزامنة السحابية للنماذج وحفظ الإجابات في نفس الوقت
  مع دعم كامل للصور لتفادي حدود سعة خلايا إكسيل (50 ألف حرف).
*/

var DATABASE_SHEET = "Cloud_Database"; // ورقة المزامنة
var RESPONSES_SHEET = "Responses_Data"; // ورقة الإجابات
var MAX_CELL_LENGTH = 49000;

function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var payloadString = e.postData.contents;
  var data = JSON.parse(payloadString);
  
  // الحالة 1: مزامنة القاعدة بالكامل (Cloud Sync)
  if (Object.keys(data).some(function(k) { return k.indexOf('formflow_') === 0; })) {
    var syncSheet = ss.getSheetByName(DATABASE_SHEET) || ss.insertSheet(DATABASE_SHEET);
    syncSheet.clear(); // مسح البيانات القديمة
    
    // تقسيم النص الطويل لعدة خلايا في العمود الأول لتخطي حاجز إكسيل
    var chunks = [];
    for (var i = 0; i < payloadString.length; i += MAX_CELL_LENGTH) {
      chunks.push([payloadString.substring(i, i + MAX_CELL_LENGTH)]);
    }
    
    if (chunks.length > 0) {
      syncSheet.getRange(1, 1, chunks.length, 1).setValues(chunks);
    }
    
    return ContentService.createTextOutput("Sync Success").setMimeType(ContentService.MimeType.TEXT);
  } 
  
  // الحالة 2: حفظ إجابة فردية (Form Submission)
  else {
    var respSheet = ss.getSheetByName(RESPONSES_SHEET) || ss.insertSheet(RESPONSES_SHEET);
    
    // إضافة العناوين إذا كانت الورقة جديدة
    if (respSheet.getLastRow() === 0) {
      respSheet.appendRow(Object.keys(data));
    }
    
    // فحص الإجابات وقص الصور الطويلة فقط داخل ورقة العرض (كي لا يتعطل الإكسيل)
    var safeValues = Object.values(data).map(function(val) {
      if (typeof val === 'string' && val.length > MAX_CELL_LENGTH) {
        return "[صورة/بيانات كبيرة مخزنة في الحفظ السحابي للوحة التحكم]";
      }
      return val;
    });
    
    respSheet.appendRow(safeValues);
    return ContentService.createTextOutput("Response Saved").setMimeType(ContentService.MimeType.TEXT);
  }
}

function doGet() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(DATABASE_SHEET);
  var data = "{}";
  
  if (sheet) {
    // سحب كل الخلايا المجزأة ودمجها مجدداً
    var values = sheet.getRange(1, 1, sheet.getLastRow() || 1, 1).getValues();
    var fullString = values.map(function(row) { return row[0]; }).join('');
    if (fullString) data = fullString;
  }
  
  return ContentService.createTextOutput(data).setMimeType(ContentService.MimeType.JSON);
}
